   $(document).ready(function() {

            let timeoutID = null;

            function findProducts(str) {

                let site_url = "https://ram.by/";

                $.ajax({
                    type: 'POST',
                    url: site_url + 'catalogsearch/ajax',
                    dataType: 'html',
                    data: {q: str},
                    success: function(data) {
                        $('.search1').find('.top-search-results').remove();
                        $('.search1').append(data);
                    },

                    error: function (xhr, ajaxOptions, thrownError) {
                        alert(xhr.status);
                        alert(thrownError);
                    }

                });
            }

            $('.search-input input[name="q"]').keyup(function(e) {
                clearTimeout(timeoutID);
                if (this.value.length > 2) {
                    timeoutID = setTimeout(() => findProducts(e.target.value), 500);
                }
            }).click(function(e) {
                if (this.value.length > 2) {
                    $('.top-search-results').show(0);
                }
            });

            $(document).mouseup(function (e) {
                let searchResults = $(".top-search-results");
                let searchInput = $('.search-input input[name="q"]');
                if (!searchInput.is(e.target) && !searchResults.is(e.target) &&
                    searchResults.has(e.target).length === 0) {
                    searchResults.hide();
                }
            });

        });
  </script>
